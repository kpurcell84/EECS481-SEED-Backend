GAE Datastore Terminology:
	Kind = Table
	Entity = Row
	Key = Primary Key
	Property = Field

#####################################

Doctor (Kind)
	Doctor (Entity)
		email (Key)
		first_name (Property)
		last_name (Property)
		phone (Property)
		hospital (Property)
	...

Patient (Kind)
	Patient (Entity)
		email (Key)
		doctor (ReferenceProperty)
		first_name (Property)
		last_name (Property)
		phone (Property)
		diagnosis (Property) - YES/NO/MAYBE
		septic_risk (Property)
	...

PQuantData (Kind)
	DataPoint (Entity)
		patient (ReferenceProperty)
		time_taken (Property)
		blood_pressure (Property)
		body_temp (Property)
		gsr (Property)
		skin_temp (Property)
		air_temp (Property)
		heart_rate (Property)
		activity_types (Property) RUN/BIKE/WALK/STILL/REM/LIGHT/DEEP
		toss_or_turn (Property) YES/NO
	...
...

PQualData (Kind)
	DataPoints (Entity)
		patient (ReferenceProperty)
		time_taken (Property)
		q1 (Property)
		q2 (Property)
		q3 (Property)
		...
	...
...	

PAlert (Kind)
	PAlert (Entity)
		patient (ReferenceProperty)
		time_alerted (Property)
		message (Property)
		priority (Property)
	...
...

WatsonQuestion (Kind)
	QuestionPair (Entity)
		question (Property)
		answer (Property)
		date_asked (Property)
	...
...
